#6

?x pr:hasInhabitantNumber ?y .

#7

select distinct ?x ?y
where {
?z rdfs:subClassOf pr:Locality .
?x rdfs:subClassOf ?z .
OPTIONAL { ?x pr:hasInhabitantNumber ?y }
} LIMIT 100

#8

select distinct ?x ?y
where {
?x rdfs:subClassOf pr:Locality .
?x pr:hasInhabitantNumber ?y .
FILTER (xsd:integer(?y) > xsd:integer("200000"))
} LIMIT 100
