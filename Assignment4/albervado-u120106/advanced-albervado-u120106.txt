A2. Dame todas las propiedades, exceptuando rdf:type, aplicables a las instancias de la clase Politician (http://dbpedia.org/ontology/Politician)

WHERE {

  ?x a <http://dbpedia.org/ontology/Politician>.

  ?x ?p ?y.

  FILTER(?p!=rdf:type)

}
